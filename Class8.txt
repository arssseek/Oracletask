CREATE OR REPLACE PACKAGE BODY logic AS 
    PROCEDURE dob 
    (
    firstn VARCHAR2,
    lastn VARCHAR2,
    ema VARCHAR2,
    p_n VARCHAR2,
    h_d DATE,
    j_id VARCHAR2,
    sal NUMBER)
    IS
    BEGIN
        SELECT MAX(employee_id) INTO e_id
        FROM employees;
        INSERT INTO employees(employee_id,first_name,last_name,email,phone_number,hire_date,job_id,salary)VALUES((e_id + 1),firstn,lastn,ema,p_n,h_d,j_id,sal);
    END dob;
    PROCEDURE udal 
    (
    nuum NUMBER
    )
    IS
    user_err EXCEPTION;
    ertx VARCHAR2(250);
    BEGIN
        SELECT MAX(employee_id) INTO e_id
        FROM employees;
        IF nuum <= e_id THEN
        DELETE FROM employees WHERE employee_id = nuum;
        ELSE RAISE user_err;
        END IF;
    EXCEPTION 
    WHEN user_err THEN 
    INSERT INTO logs VALUES(SYSDATE,2);
    SELECT err_tx INTO ertx FROM err WHERE err_c = 2 AND err_lan = err_lang;
    Dbms_output.put_line(ertx);
    END udal;
    PROCEDURE obnov
    (
    nsal NUMBER,
    nuum NUMBER
    )
    IS
    BEGIN
        UPDATE employees
        SET salary = nsal
        WHERE employee_id = nuum;
    END obnov;
    PROCEDURE perev
     (
    ndep NUMBER,
    nuum NUMBER
    )
    IS
    BEGIN
        UPDATE employees
        SET department_id = ndep
        WHERE employee_id = nuum;
    END perev;
    FUNCTION dep(
    dep_id NUMBER)
    RETURN VARCHAR2
    IS
    dep_name VARCHAR2(15);
    dep_err EXCEPTION;
    ertx VARCHAR2(250);
    BEGIN
        SELECT DISTINCT department_name INTO dep_name FROM departments
        WHERE department_id = dep_id;
        IF dep_name <> '' THEN
        RETURN dep_name;
        ELSE RAISE dep_err;
        END IF;
    EXCEPTION
    WHEN dep_err THEN 
    INSERT INTO logs VALUES(SYSDATE,1);
    SELECT err_tx INTO ertx FROM err WHERE err_c = 1 AND err_lan = err_lang;
    Dbms_output.put_line(ertx);
    END dep;
END logic;
